Book Information System

#include 
struct Book { char title[50]; char author[30]; float price; };

void inputBook(struct Book *b) {
    printf("Enter Title: "); scanf("%s", b->title);
    printf("Enter Author: "); scanf("%s", b->author);
    printf("Enter Price: "); scanf("%f", &b->price);
}

void displayBook(struct Book *b) { printf("Book: %s | Author: %s | Price: %.2f\n", b->title, b->author, b->price); }

int main() {
    struct Book b;
    inputBook(&b);
    displayBook(&b);
    return 0;
}
