Student Marks System

#include
struct Student { int marks; };
void initStudent(struct Student *s, int marks) { s->marks = marks; }
void calculateAverage(struct Student *s) {
printf("Total marks after bonus: %d\n", s->marks + 10);
}
void resetStudent(struct Student *s) { s->marks = 100; printf("Student reset!\n"); }
int main() {
struct Student s;
initStudent(&s, 70); calculateAverage(&s);
resetStudent(&s);
calculateAverage(&s);
return 0;
}
