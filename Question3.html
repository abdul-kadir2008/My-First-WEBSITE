Rectangle Area & Perimeter

#include
struct Rectangle { int length, breadth; };
void initRectangle(struct Rectangle *r, int l, int b) { r->length = l; r->breadth = b; }
int area(struct Rectangle *r) { return r->length * r->breadth; }
int perimeter(struct Rectangle *r) { return 2 * (r->length + r->breadth); }
int main() {
struct Rectangle r;
initRectangle(&r, 10, 5);
printf("Area: %d\n", area(&r));
printf("Perimeter: %d\n", perimeter(&r));
return 0;
}
