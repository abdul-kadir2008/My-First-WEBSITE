Employee Salary System

#include
struct Employee { int id; float salary; };
void initEmployee(struct Employee *e, int id, float salary) { e->id = id; e->salary = salary; }
void giveBonus(struct Employee *e, float bonus) {
e->salary += bonus;
printf("New Salary: %.2f\n", e->salary);
}
void display(struct Employee *e) { printf("Employee ID: %d | Salary: %.2f\n", e->id, e->salary); }
int main() {
struct Employee e;
initEmployee(&e, 101, 25000);
display(&e);
giveBonus(&e, 5000);
return 0;
}
